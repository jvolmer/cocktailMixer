class Mover {
    {field} Coordinate position
    {field} Direction direction
    void followLine()
    void followLineUntilCrossing()
    void turnRightAtCrossing()
    void turnLeftAtCrossing()
}

class Tracker {
  {field} RoadLayout roadLayout
  RoadLayout checkRoad()
}

class Motor {
  byte leftSpeed
  byte rightSpeed
  void run(leftSpeed, rightSpeed)
  void turnLeft()
  void turnRight()
  void goStraight()
  void stop()
}

enum RoadLayout {
  none
  blocked
  enclosed
  straight
  sharpLeft
  left
  sharpRight
  right
}
class Coordinate {
  int x
  int y
}
enum Direction {
  positiveX
  negativeY
  negativeX
  positiveY
}


Mover o-- Motor
Mover o-- Tracker
Mover *-- Coordinate
Mover *-- Direction
Tracker *-- RoadLayout


@enduml
